\pgfkeys{
  /businessbubble/.is family, /businessbubble,
  default/.style = { color = blue, opacity = 1, x = 0, y = 0, size = 4cm, textcolor = white},
  color/.estore in = \bcolor,
  opacity/.estore in = \bopacity,
  x/.estore in = \bx, y/.estore in = \by,
  size/.estore in = \bsize, textcolor/.estore in = \btextcol,
}

\newcommand{\bubbleexport}[2][]{%
  \pgfkeys{/businessbubble, default, #1}
  \pgfmathsetmacro{\btextwidth}{\bsize * 0.72}
  \node[
    circle, fill=\bcolor, fill opacity=\bopacity, text opacity=1,
    minimum size=\bsize, text width=\btextwidth pt, text=\btextcol,
    font=\rmfamily\fontsize{10}{12}\selectfont, align=center, inner sep=10pt
  ] at (\bx, \by) {#2};
}
